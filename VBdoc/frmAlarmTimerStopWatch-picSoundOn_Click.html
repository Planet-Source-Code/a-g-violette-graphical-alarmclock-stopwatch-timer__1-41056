Private Sub picSoundOn_Click()
  
  ' When Sound On image is clicked change to Sound Off image and turn off ticker
  '
  picSoundOn.Visible = False
  picSoundOff.Visible = True
  MediaPlayer1.Stop
  MediaPlayer1.FastReverse

End Sub
